#!/bin/bash
sudo apt update -y 
sudo apt install -y gnupg software-properties-common wget
wget -q -O - https://pkg.jenkins.io/debian/jenkins.io-2023.key | sudo tee /usr/share/keyrings/jenkins-keyring.asc > /dev/null
echo "deb [signed-by=/usr/share/keyrings/jenkins-keyring.asc] https://pkg.jenkins.io/debian-stable binary/" | sudo tee /etc/apt/sources.list.d/jenkins.list
sudo apt update -y
sudo apt install -y openjdk-17-jdk jenkins python3-pip
sudo systemctl enable jenkins
sudo systemctl start jenkins
