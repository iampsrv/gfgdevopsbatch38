kubectl create deployment nginx --image=nginx:1.25
kubectl expose deployment nginx --port=80 --target-port=80
kubectl port-forward svc/nginx 8080:80
kubectl expose deployment nginx --type=NodePort --name=nginx-node --port=80 --target-port=80
kubectl get svc nginx-node -o wide
kubectl expose deployment nginx --type=LoadBalancer --name=nginx-lb --port=80 --target-port=80
kubectl get svc nginx-lb


kubectl run nginx --image=nginx:1.25 --port=80
kubectl expose pod nginx --port=80 --target-port=80 --type=NodePort
